def prettifyDate(dt):
    try:
        dt_datetime = datetime.strptime(dt, '%d/%m/%Y %H:%M')
        dt_string = datetime.strftime(dt_datetime, '%Y-%m-%d %H:%M')
        return dt_string
    except:
        return np.nan
